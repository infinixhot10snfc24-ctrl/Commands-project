repeat task.wait() until getgenv().DeltaCmds and getgenv().DeltaCmds.Cmds
local p = game.Players.LocalPlayer
local rs = game:GetService("RunService")
local uis = game:GetService("UserInputService")
local lighting = game:GetService("Lighting")
local cam = workspace.CurrentCamera
local vwConn = nil
local viewRemoveConn = nil
local fcConn, fcInputConn, fcCharConn = nil, nil, nil
local fcActive = false
local freecamCF = nil
local antiflingConn = nil
local isMobile = uis.TouchEnabled and not uis.KeyboardEnabled
local touchRotStart = nil
local function stopVw()
	if vwConn then vwConn:Disconnect(); vwConn = nil end
	if viewRemoveConn then viewRemoveConn:Disconnect(); viewRemoveConn = nil end
	local char = p.Character
	local hum = char and char:FindFirstChildOfClass("Humanoid")
	if hum then
		cam.CameraSubject = hum
		cam.CameraType = Enum.CameraType.Custom
	end
end
local function startVw(name)
	stopVw()
	local target = getgenv().DeltaCmds.resolveTargets(name)[1]
	if not target then return end
	local function applySubject()
		local char = target.Character
		local hum = char and char:FindFirstChildOfClass("Humanoid")
		if hum then
			cam.CameraSubject = hum
			cam.CameraType = Enum.CameraType.Custom
		end
	end
	applySubject()
	vwConn = target.CharacterAdded:Connect(function()
		task.wait(0.1)
		applySubject()
	end)
	viewRemoveConn = game.Players.PlayerRemoving:Connect(function(pl)
		if pl == target then stopVw() end
	end)
end
local _up = Vector3.new(0,1,0)
local function stopFc()
	fcActive = false
	if fcConn then fcConn:Disconnect(); fcConn = nil end
	if fcCharConn then fcCharConn:Disconnect(); fcCharConn = nil end
	if fcInputConn then fcInputConn:Disconnect(); fcInputConn = nil end
	touchRotStart = nil
	cam.CameraType = Enum.CameraType.Custom
	local char = p.Character
	if not char then return end
	local hum = char:FindFirstChildOfClass("Humanoid")
	if hum then cam.CameraSubject = hum end
	local root = char:FindFirstChild("HumanoidRootPart")
	if root then
		local bv = root:FindFirstChild("FreecamFreeze")
		if bv then bv:Destroy() end
	end
end
local function startFc(spd)
	stopFc()
	fcActive = true
	cam.CameraType = Enum.CameraType.Scriptable
	freecamCF = cam.CFrame
	local speed = spd or 1
	local function applyFreezeBV(char)
		local root = char and char:FindFirstChild("HumanoidRootPart")
		if root then
			local bv = Instance.new("BodyVelocity", root)
			bv.Name = "FreecamFreeze"
			bv.Velocity = Vector3.zero
			bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
		end
	end
	if p.Character then applyFreezeBV(p.Character) end
	fcCharConn = p.CharacterAdded:Connect(applyFreezeBV)
	fcConn = rs.RenderStepped:Connect(function(dt)
		if not fcActive then return end
		cam.CameraType = Enum.CameraType.Scriptable
		local cf = freecamCF
		local move = Vector3.zero
		if isMobile then
			local spd2 = speed
			local mv = getgenv().DeltaCmds.ctrls:GetMoveVector()
			if mv.Magnitude > 0.01 then
				move += cf.LookVector * -mv.Z + cf.RightVector * mv.X
			end
			if move.Magnitude > 0 then
				freecamCF = CFrame.new(cf.Position + move.Unit * spd2 * 60 * dt) * (cf - cf.Position)
			end
		else
			local spd2 = speed * (uis:IsKeyDown(Enum.KeyCode.LeftShift) and 3 or 1)
			if uis:IsKeyDown(Enum.KeyCode.W) then move += cf.LookVector end
			if uis:IsKeyDown(Enum.KeyCode.S) then move -= cf.LookVector end
			if uis:IsKeyDown(Enum.KeyCode.A) then move -= cf.RightVector end
			if uis:IsKeyDown(Enum.KeyCode.D) then move += cf.RightVector end
			if uis:IsKeyDown(Enum.KeyCode.E) or uis:IsKeyDown(Enum.KeyCode.Space) then move += _up end
			if uis:IsKeyDown(Enum.KeyCode.Q) then move -= _up end
			if move.Magnitude > 0 then
				freecamCF = CFrame.new(cf.Position + move.Unit * spd2 * 60 * dt) * (cf - cf.Position)
			end
		end
		cam.CFrame = freecamCF
	end)
	if isMobile then
		local screenSize = cam.ViewportSize
		local rightHalf = screenSize.X / 2
		fcInputConn = uis.InputChanged:Connect(function(input)
			if not fcActive then return end
			if input.UserInputType == Enum.UserInputType.Touch then
				local pos = input.Position
				if pos.X >= rightHalf then
					if touchRotStart then
						local dx = -(pos.X - touchRotStart.X) * 0.004
						local dy = -(pos.Y - touchRotStart.Y) * 0.004
						local cf = freecamCF
						freecamCF = CFrame.new(cf.Position)
							* CFrame.Angles(0, dx, 0)
							* (cf - cf.Position)
							* CFrame.Angles(dy, 0, 0)
						touchRotStart = Vector2.new(pos.X, pos.Y)
					end
				end
			end
		end)
		local touchStartConn = uis.TouchStarted:Connect(function(input)
			if not fcActive then return end
			local pos = input.Position
			if pos.X >= rightHalf then
				touchRotStart = Vector2.new(pos.X, pos.Y)
			end
		end)
		local touchEndConn = uis.TouchEnded:Connect(function(input)
			if not fcActive then return end
			local pos = input.Position
			if pos.X >= rightHalf then
				touchRotStart = nil
			end
		end)
		local oldInputConn = fcInputConn
		fcInputConn = {
			Disconnect = function()
				if oldInputConn then oldInputConn:Disconnect() end
				touchStartConn:Disconnect()
				touchEndConn:Disconnect()
			end
		}
	else
		fcInputConn = uis.InputChanged:Connect(function(input)
			if not fcActive then return end
			if input.UserInputType == Enum.UserInputType.MouseMovement then
				local dx = -input.Delta.X * 0.003
				local dy = -input.Delta.Y * 0.003
				local cf = freecamCF
				freecamCF = CFrame.new(cf.Position)
					* CFrame.Angles(0, dx, 0)
					* (cf - cf.Position)
					* CFrame.Angles(dy, 0, 0)
			end
		end)
	end
end
local origFog = lighting.FogEnd
local nfActive = false
local nfConn = nil
local function startNf()
	origFog = lighting.FogEnd
	nfActive = true
	lighting.FogEnd = 1e9
	if nfConn then nfConn:Disconnect() end
	nfConn = lighting:GetPropertyChangedSignal("FogEnd"):Connect(function()
		if nfActive and lighting.FogEnd < 1e9 then
			lighting.FogEnd = 1e9
		end
	end)
end
local function stopNf()
	if not nfActive then return end
	nfActive = false
	if nfConn then nfConn:Disconnect(); nfConn = nil end
	lighting.FogEnd = origFog
end
local aflConns = {}
local aflOrig = {}
local function stopAfl()
	for _, conn in ipairs(aflConns) do conn:Disconnect() end
	aflConns = {}
	if antiflingConn then antiflingConn:Disconnect(); antiflingConn = nil end
	for part, orig in pairs(aflOrig) do
		if part and part.Parent then part.CanCollide = orig end
	end
	aflOrig = {}
end
local function startAfl()
	stopAfl()
	local function applyPart(part)
		if not aflOrig[part] then
			aflOrig[part] = part.CanCollide
			part.AncestryChanged:Connect(function()
				if not part.Parent then
					aflOrig[part] = nil
				end
			end)
		end
		part.CanCollide = false
	end
	local function applyChar(char)
		for _, part in ipairs(char:GetDescendants()) do
			if part:IsA("BasePart") then applyPart(part) end
		end
		aflConns[#aflConns+1] = char.DescendantAdded:Connect(function(part)
			if part:IsA("BasePart") then applyPart(part) end
		end)
	end
	local function onPlayer(pl)
		if pl == p then return end
		if pl.Character then applyChar(pl.Character) end
		aflConns[#aflConns+1] = pl.CharacterAdded:Connect(applyChar)
	end
	for _, pl in ipairs(game.Players:GetPlayers()) do onPlayer(pl) end
	antiflingConn = game.Players.PlayerAdded:Connect(onPlayer)
end
local frzConn = nil
local function stopFrz()
	if frzConn then frzConn:Disconnect(); frzConn = nil end
	local char = p.Character
	if not char then return end
	for _, v in ipairs(char:GetDescendants()) do
		if v:IsA("BasePart") then v.Anchored = false end
	end
end
local function startFrz()
	stopFrz()
	local function applyFreeze(char)
		for _, v in ipairs(char:GetDescendants()) do
			if v:IsA("BasePart") then v.Anchored = true end
		end
	end
	local char = p.Character or p.CharacterAdded:Wait()
	applyFreeze(char)
	frzConn = p.CharacterAdded:Connect(applyFreeze)
end
getgenv().DeltaCmds.startFc = startFc
getgenv().DeltaCmds.stopFc  = stopFc
getgenv().DeltaCmds.startFrz  = startFrz
getgenv().DeltaCmds.stopFrz   = stopFrz

local tps = game:GetService("TeleportService")
local hs = game:GetService("HttpService")
local akActive = false
local akOrig = nil
local function startAk()
	if akActive then return end
	akActive = true
	local m = getrawmetatable(game)
	setreadonly(m, false)
	akOrig = m.__namecall
	m.__namecall = newcclosure(function(s, ...)
		if tostring(getnamecallmethod()) == "Kick" then
			task.wait(9e9)
			return
		end
		return akOrig(s, ...)
	end)
	setreadonly(m, true)
end
local function stopAk()
	if not akActive then return end
	local m = getrawmetatable(game)
	setreadonly(m, false)
	m.__namecall = akOrig
	setreadonly(m, true)
	akOrig = nil
	akActive = false
end

local function serverhop()
	local placeId = game.PlaceId
	local curJob = game.JobId
	local servers = {}
	local cursor = ""
	repeat
		local url = "https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?sortOrder=Asc&limit=100"
		if cursor ~= "" then url = url .. "&cursor=" .. cursor end
		local ok, res = pcall(function() return game:HttpGet(url) end)
		if not ok then break end
		local data = hs:JSONDecode(res)
		if not data or not data.data then break end
		for _, s in ipairs(data.data) do
			if s.id ~= curJob and s.playing < s.maxPlayers then
				servers[#servers+1] = s
			end
		end
		cursor = data.nextPageCursor or ""
	until cursor == ""
	if #servers == 0 then return end
	local pick = servers[math.random(#servers)]
	tps:TeleportToPlaceInstance(placeId, pick.id, p)
end

local function sshop_fn()
	local placeId = game.PlaceId
	local curJob = game.JobId
	local cursor = ""
	local best = nil
	repeat
		local url = "https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?sortOrder=Asc&limit=100"
		if cursor ~= "" then url = url .. "&cursor=" .. cursor end
		local ok, res = pcall(function() return game:HttpGet(url) end)
		if not ok then break end
		local data = hs:JSONDecode(res)
		if not data or not data.data then break end
		for _, s in ipairs(data.data) do
			if s.id ~= curJob and s.playing < s.maxPlayers then
				if not best or s.playing < best.playing then
					best = s
				end
			end
		end
		cursor = data.nextPageCursor or ""
	until cursor == "" or best and best.playing <= 1
	if not best then return end
	tps:TeleportToPlaceInstance(placeId, best.id, p)
end

local alGui = nil
local alHB = nil

local function stopAl()
	R:UnbindFromRenderStep("DeltaAimlock")
	if alHB then alHB:Disconnect(); alHB = nil end
	if alGui then alGui:Destroy(); alGui = nil end
end

local function startAl(fov)
	stopAl()
	local F = fov or 120
	local F2 = F*F
	local A = 0.3
	local S = 0.015
	local M = 1000
	local PR = 0.08
	local L = p
	local C = cam
	local G = L:WaitForChild("PlayerGui")
	local g = Instance.new("ScreenGui")
	g.IgnoreGuiInset = true
	g.ResetOnSpawn = false
	g.Parent = G
	alGui = g
	local c = Instance.new("Frame")
	c.AnchorPoint = Vector2.new(.5,.5)
	c.Position = UDim2.new(.5,0,.5,0)
	c.Size = UDim2.fromOffset(F*2,F*2)
	c.BackgroundTransparency = 1
	c.Parent = g
	local u = Instance.new("UICorner")
	u.CornerRadius = UDim.new(1,0)
	u.Parent = c
	local s = Instance.new("UIStroke")
	s.Thickness = 2
	s.Color = Color3.fromRGB(255,255,255)
	s.Parent = c
	local t = nil
	local d = 0
	local cp = {}
	for _,pl in ipairs(game.Players:GetPlayers()) do
		if pl ~= L then cp[#cp+1] = pl end
	end
	game.Players.PlayerAdded:Connect(function(pl)
		if pl ~= L then cp[#cp+1] = pl end
	end)
	game.Players.PlayerRemoving:Connect(function(pl)
		for i=#cp,1,-1 do
			if cp[i]==pl then table.remove(cp,i) break end
		end
	end)
	local rp = RaycastParams.new()
	rp.FilterType = Enum.RaycastFilterType.Blacklist
	local function v(h)
		local o = C.CFrame.Position
		local dir = h.Position-o
		rp.FilterDescendantsInstances = {L.Character,h.Parent}
		local r = workspace:Raycast(o,dir,rp)
		if not r then return true end
		return false
	end
	local function ok(h)
		if not h or not h.Parent then return false end
		local ch = h.Parent
		local hu = ch:FindFirstChild("Humanoid")
		local ro = ch:FindFirstChild("HumanoidRootPart")
		if not hu or hu.Health<=0 then return false end
		local my = L.Character
		local mr = my and my:FindFirstChild("HumanoidRootPart")
		if not mr then return false end
		if (ro.Position-mr.Position).Magnitude>M then return false end
		local pos,vis = C:WorldToViewportPoint(h.Position)
		if not vis then return false end
		local cx = C.ViewportSize.X*.5
		local cy = C.ViewportSize.Y*.5
		local dx = pos.X-cx
		local dy = pos.Y-cy
		if dx*dx+dy*dy>F2 then return false end
		if not v(h) then return false end
		return true
	end
	local function gcl()
		local cl = nil
		local cd = F2
		local cx = C.ViewportSize.X*.5
		local cy = C.ViewportSize.Y*.5
		local my = L.Character
		local mr = my and my:FindFirstChild("HumanoidRootPart")
		if not mr then return nil end
		for i=1,#cp do
			local pl = cp[i]
			local ch = pl.Character
			if ch then
				local h = ch:FindFirstChild("Head")
				local hu = ch:FindFirstChild("Humanoid")
				local ro = ch:FindFirstChild("HumanoidRootPart")
				if h and hu and hu.Health>0 and ro then
					if (ro.Position-mr.Position).Magnitude<=M then
						local pos,vis = C:WorldToViewportPoint(h.Position)
						if vis and v(h) then
							local dx = pos.X-cx
							local dy = pos.Y-cy
							local dist = dx*dx+dy*dy
							if dist<cd then cd=dist cl=h end
						end
					end
				end
			end
		end
		return cl
	end
	alHB = rs.Heartbeat:Connect(function(dt)
		d+=dt
		if d>S then d=0
			if not ok(t) then t=gcl() end
		end
	end)
	R:BindToRenderStep("DeltaAimlock",Enum.RenderPriority.Last.Value,function()
		if t and t.Parent then
			local ro = t.Parent:FindFirstChild("HumanoidRootPart")
			local pos = t.Position
			if ro then
				local vv = ro.AssemblyLinearVelocity
				pos = pos+Vector3.new(vv.X,0,vv.Z)*PR
			end
			local cf = CFrame.new(C.CFrame.Position,pos)
			C.CFrame = C.CFrame:Lerp(cf,A)
		end
	end)
end
