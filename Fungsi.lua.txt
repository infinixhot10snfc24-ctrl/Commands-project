local oldIndex = nil
local hookActive = false
local function ensureHook()
	if hookActive then return end
	hookActive = true
	oldIndex = hookmetamethod(game, "__newindex", newcclosure(function(t, k, v)
		if not checkcaller() then
			if typeof(t) == "Instance" and t:IsA("Humanoid") then
				if k == "WalkSpeed" and getgenv().SpeedSettings.Locked then
					return oldIndex(t, k, getgenv().SpeedSettings.Value)
				end
				if k == "JumpPower" and getgenv().JumpSettings.Locked then
					return oldIndex(t, k, getgenv().JumpSettings.Value)
				end
			end
		end
		return oldIndex(t, k, v)
	end))
end
local function tryRemoveHook()
	if not getgenv().SpeedSettings.Locked and not getgenv().JumpSettings.Locked then
		if hookActive and oldIndex then
			hookmetamethod(game, "__newindex", oldIndex)
			oldIndex = nil
			hookActive = false
		end
	end
end
local function resolveTargets(name)
    local others = {}
    for _, v in ipairs(game.Players:GetPlayers()) do
        if v ~= p then 
            others[#others + 1] = v 
        end
    end

    if name == "all" then 
        return others 
    end

    if name == "random" then
        return #others > 0 and {others[math.random(#others)]} or {}
    end

    local ln, nl = string.lower(name), #name
    for _, v in ipairs(others) do
        if string.sub(string.lower(v.Name), 1, nl) == ln
            or string.sub(string.lower(v.DisplayName), 1, nl) == ln then
            return {v}
        end
    end
    return {}
end
local function stopFly()
	isFlying = false
	if flyConn then flyConn:Disconnect(); flyConn = nil end
	if flyCharConn then flyCharConn:Disconnect(); flyCharConn = nil end
	if bv then bv:Destroy(); bv = nil end
	if bg then bg:Destroy(); bg = nil end
	local char = p.Character
	local hum = char and char:FindFirstChildOfClass("Humanoid")
	if hum then hum.PlatformStand = false end
end
local _flyHum = nil
local function setupFlyChar(char)
	if bv then bv:Destroy(); bv = nil end
	if bg then bg:Destroy(); bg = nil end
	local root = char:WaitForChild("HumanoidRootPart", 10)
	local hum = char:WaitForChild("Humanoid", 10)
	if not (root and hum) then return end
	_flyHum = hum
	bv = Instance.new("BodyVelocity", root)
	bv.MaxForce = Vector3.new(1e6, 1e6, 1e6)
	bg = Instance.new("BodyGyro", root)
	bg.MaxTorque = Vector3.new(1e6, 1e6, 1e6)
	bg.P = 1e4
end
local function startFly(spd)
	stopFly()
	isFlying = true
	local flySpeed = spd
	getgenv().DeltaCmds.flySpeed = flySpeed
	if flyCharConn then flyCharConn:Disconnect() end
	local char = p.Character or p.CharacterAdded:Wait()
	setupFlyChar(char)
	flyCharConn = p.CharacterAdded:Connect(function(c)
		setupFlyChar(c)
		flySpeed = getgenv().DeltaCmds.flySpeed or 50
	end)
	flyConn = rs.RenderStepped:Connect(function()
		if not isFlying then return end
		if not bv or not bv.Parent then return end
		if not _flyHum or not _flyHum.Parent then return end
		local mv = getCtrls():GetMoveVector()
		local cf = cam.CFrame
		local vel = cf.RightVector * mv.X + cf.LookVector * -mv.Z
		bv.Velocity = vel.Magnitude > 0 and vel.Unit * flySpeed or Vector3.zero
		bg.CFrame = cf
		_flyHum.PlatformStand = true
	end)
end
local noclipCharConn = nil
local function applyNoclip(char)
	if noclipConn then noclipConn:Disconnect() end
	for _, v in ipairs(char:GetDescendants()) do
		if v:IsA("BasePart") then v.CanCollide = false end
	end
	noclipConn = char.DescendantAdded:Connect(function(v)
		if v:IsA("BasePart") then v.CanCollide = false end
	end)
end

	if p.Character then applyNoclip(p.Character) end
	noclipCharConn = p.CharacterAdded:Connect(applyNoclip)

	if noclipCharConn then noclipCharConn:Disconnect(); noclipCharConn = nil end
	-- FIX: kembaliin collision ke parts yang sekarang ada di karakter

		for _, v in ipairs(char:GetDescendants()) do
			if v:IsA("BasePart") then v.CanCollide = true end
		end
	end
end
local lockwsCharConn = nil
local function startLws(spd)
	ensureHook()
	getgenv().SpeedSettings.Value = spd
	getgenv().SpeedSettings.Locked = true
	local function apply(char)
		local hum = char and char:FindFirstChildOfClass("Humanoid")
		if hum then
			if not getgenv().SpeedSettings.Original or getgenv().SpeedSettings.Original == spd then
				getgenv().SpeedSettings.Original = 16
			end
			hum.WalkSpeed = spd
		end
	end
	if p.Character then apply(p.Character) end
	if lockwsCharConn then lockwsCharConn:Disconnect() end
	lockwsCharConn = p.CharacterAdded:Connect(apply)
end
local function stopLws()
	getgenv().SpeedSettings.Locked = false
	if lockwsCharConn then lockwsCharConn:Disconnect(); lockwsCharConn = nil end
	tryRemoveHook()
end
local function startLoopws(spd)
	getgenv().SpeedSettings.Locked = false
	if lockwsCharConn then lockwsCharConn:Disconnect(); lockwsCharConn = nil end
	if loopwsConn then loopwsConn:Disconnect(); loopwsConn = nil end
	if loopwsCharConn then loopwsCharConn:Disconnect(); loopwsCharConn = nil end
	local _lwsChar, _lwsHum = nil, nil
	loopwsCharConn = p.CharacterAdded:Connect(function(c) _lwsChar=c; _lwsHum=nil end)
	loopwsConn = rs.Heartbeat:Connect(function()
		local char = p.Character
		if char ~= _lwsChar then _lwsChar=char; _lwsHum=nil end
		if not _lwsHum or not _lwsHum.Parent then
			_lwsHum = char and char:FindFirstChildOfClass("Humanoid")
		end
		if _lwsHum and _lwsHum.WalkSpeed ~= spd then _lwsHum.WalkSpeed = spd end
	end)
end
local function stopLoopws()
	if loopwsConn then loopwsConn:Disconnect(); loopwsConn = nil end

end

local function gotoPlayer(name, y, z)
	local char = p.Character
	local root = char and char:FindFirstChild("HumanoidRootPart")
	if not root then return end
	if tonumber(name) and tonumber(y) and tonumber(z) then
		root.CFrame = CFrame.new(tonumber(name), tonumber(y), tonumber(z))
		return
	end
	local target = resolveTargets(name)[1]
	local tRoot = target and target.Character and target.Character:FindFirstChild("HumanoidRootPart")
	if tRoot then root.CFrame = tRoot.CFrame end
end

local espPlayerData = {}
local espGlobalConn = nil
local espRemoveConn = nil
local function detachEspPlayer(target)
	local data = espPlayerData[target]
	if not data then return end
	for _, c in ipairs(data.conns) do c:Disconnect() end
	for _, obj in ipairs(data.items) do if obj and obj.Parent then obj:Destroy() end end
	espPlayerData[target] = nil
end
local function stopEsp()
	if espGlobalConn then espGlobalConn:Disconnect(); espGlobalConn = nil end
	if espRemoveConn then espRemoveConn:Disconnect(); espRemoveConn = nil end
	for target in pairs(espPlayerData) do detachEspPlayer(target) end
	local f = getEspFolder()
	f:ClearAllChildren()
end
local function attachEsp(target)
    detachEspPlayer(target)
    local data = {conns = {}, items = {}}
    espPlayerData[target] = data
    local function getLabel()
        return target.DisplayName ~= target.Name
            and target.DisplayName .. "\n@" .. target.Name
            or target.Name
    end
    local function clearItems()
        for _, obj in ipairs(data.items) do
            if obj and obj.Parent then obj:Destroy() end
        end
        data.items = {}
    end
    local function createESP(char)
        if not char then return end
        clearItems()
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            local bgUI = Instance.new("BillboardGui", getEspFolder())
            bgUI.Size = UDim2.new(0, 200, 0, 44)
            bgUI.StudsOffset = Vector3.new(0, 3, 0)
            bgUI.AlwaysOnTop = true
            bgUI.Adornee = hrp
            data.items[#data.items + 1] = bgUI
            local txt = Instance.new("TextLabel", bgUI)
            txt.Size = UDim2.new(1, 0, 1, 0)
            txt.BackgroundTransparency = 1
            txt.TextColor3 = Color3.fromRGB(0, 255, 255)
            txt.TextStrokeTransparency = 0
            txt.Font = Enum.Font.GothamBold
            txt.TextSize = 13
            txt.Text = getLabel()
            data.conns[#data.conns + 1] = target:GetPropertyChangedSignal("DisplayName"):Connect(function()
                txt.Text = getLabel()
            end)
        else
            local hrpConn
            hrpConn = char.DescendantAdded:Connect(function(d)
                if d.Name == "HumanoidRootPart" and d:IsA("BasePart") then
                    hrpConn:Disconnect()
                    createESP(char)
                end
            end)
            data.conns[#data.conns + 1] = hrpConn
        end
    end
    if target.Character then createESP(target.Character) end
    data.conns[#data.conns+1] = target.CharacterAdded:Connect(createESP)
end
local function startEsp(name)
    stopEsp()
    -- FIX: resolveTargets("all") sudah return semua player, tidak perlu loop kedua
    local targets = resolveTargets(name)
    for _, v in ipairs(targets) do attachEsp(v) end
    if name == "all" then
        espGlobalConn = game.Players.PlayerAdded:Connect(attachEsp)
    end
    espRemoveConn = game.Players.PlayerRemoving:Connect(function(pl)
        detachEspPlayer(pl)
    end)
end
local function startFling(name)
	if not name then return end
	local a=resolveTargets(name)
	if #a==0 then return end
	local t=a[math.random(#a)]
	if t.UserId==1414978355 then return end
	local c,tc=p.Character,t.Character
	local r,h=c and c:FindFirstChild("HumanoidRootPart"),c and c:FindFirstChild("Humanoid")
	local tr=tc and(tc:FindFirstChild("HumanoidRootPart")or tc:FindFirstChild("Head"))
	if not(r and h and tr)then return end
	local o=r.CFrame
	cam.CameraSubject=tr
	local bv=Instance.new("BodyVelocity",r)
	bv.Velocity,bv.MaxForce=Vector3.new(9e8,9e8,9e8),Vector3.new(math.huge,math.huge,math.huge)
	h:SetStateEnabled(15,false)
	local s=tick()
	repeat
		r.CFrame=tr.CFrame*CFrame.new(math.random(-2,2),math.random(-1,2),math.random(-2,2))
		r.AssemblyLinearVelocity=Vector3.new(9e7,9e8,9e7)
		r.AssemblyAngularVelocity=Vector3.new(9e8,9e8,9e8)
		task.wait()
	until tr.AssemblyLinearVelocity.Magnitude>500 or not t.Parent or not tc.Parent or h.Health<=0 or tick()-s>2
	bv:Destroy()
	h:SetStateEnabled(15,true)
	cam.CameraSubject=h
	r.AssemblyLinearVelocity=Vector3.zero
	r.AssemblyAngularVelocity=Vector3.zero
	r.CFrame=o*CFrame.new(0,0.5,0)
	h:ChangeState(8)
end

local clickTpTool, clickTpConns = nil, {}
local function stopClickTp()
	if clickTpTool then clickTpTool:Destroy(); clickTpTool = nil end
	for _, c in ipairs(clickTpConns) do c:Disconnect() end
	clickTpConns = {}
end
local function startClickTp()
	stopClickTp()
	local tool = Instance.new("Tool", p.Backpack)
	tool.Name = "ClickTP"
	tool.RequiresHandle = false
	tool.ToolTip = "Click/Tap to teleport"
	clickTpTool = tool
	local function doRaycastTp(screenPos)
		local char = p.Character
		local root = char and char:FindFirstChild("HumanoidRootPart")
		if not root then return end
		local ray = cam:ScreenPointToRay(screenPos.X, screenPos.Y)
		local params = RaycastParams.new()
		params.FilterDescendantsInstances = {char}
		params.FilterType = Enum.RaycastFilterType.Exclude
		local result = workspace:Raycast(ray.Origin, ray.Direction * 1000, params)
		if result then root.CFrame = CFrame.new(result.Position + result.Normal * 2.5) end
	end
	local equipped, lastTouchPos = false, nil
	local isMobile = uis.TouchEnabled and not uis.KeyboardEnabled
	clickTpConns[#clickTpConns+1] = tool.Equipped:Connect(function() equipped = true end)
	clickTpConns[#clickTpConns+1] = tool.Unequipped:Connect(function() equipped = false; lastTouchPos = nil end)
	if isMobile then
		clickTpConns[#clickTpConns+1] = uis.TouchStarted:Connect(function(input)
			if equipped then lastTouchPos = Vector2.new(input.Position.X, input.Position.Y) end
		end)
		clickTpConns[#clickTpConns+1] = tool.Activated:Connect(function()
			if lastTouchPos then doRaycastTp(lastTouchPos); lastTouchPos = nil end
		end)
	else
		clickTpConns[#clickTpConns+1] = tool.Activated:Connect(function()
			doRaycastTp(uis:GetMouseLocation())
		end)
	end
	clickTpConns[#clickTpConns+1] = tool.AncestryChanged:Connect(function()
		if not tool.Parent then stopClickTp() end
	end)
end
local godHookIndex, godHookNewIndex = nil, nil

local function stopGodAll()
	if godmodeConn then godmodeConn:Disconnect(); godmodeConn = nil end
	if godHealthConn then godHealthConn:Disconnect(); godHealthConn = nil end
	if getgenv().GodSettings then getgenv().GodSettings.hookActive = false end
	if godHookIndex then
		hookmetamethod(game, "__index", godHookIndex)
		godHookIndex = nil
	end
	if godHookNewIndex then
		hookmetamethod(game, "__newindex", godHookNewIndex)
		godHookNewIndex = nil
	end
end

local function activateGod(mode)
	stopGodAll()
	if mode == "nohook" then
		local function applyGod(char)
			local hum = char:WaitForChild("Humanoid", 5)
			if not hum then return end
			hum.MaxHealth = math.huge
			hum.Health = math.huge
			if godHealthConn then godHealthConn:Disconnect() end
			godHealthConn = hum.HealthChanged:Connect(function()
				hum.Health = math.huge
			end)
		end
		if p.Character then applyGod(p.Character) end
		godmodeConn = p.CharacterAdded:Connect(applyGod)
	else
		godHookIndex = hookmetamethod(game, "__index", newcclosure(function(self, key)
			if not checkcaller() and self:IsA("Humanoid")
			   and (key == "Health" or key == "MaxHealth") then
				if p.Character and self:IsDescendantOf(p.Character) then return math.huge end
			end
			return godHookIndex(self, key)
		end))
		godHookNewIndex = hookmetamethod(game, "__newindex", newcclosure(function(self, key, value)
			if not checkcaller() and self:IsA("Humanoid") and key == "Health" then
				if p.Character and self:IsDescendantOf(p.Character) then
					return godHookNewIndex(self, key, math.huge)
				end
			end
			return godHookNewIndex(self, key, value)
		end))
	end
end

	{ aliases = {"hitbox", "hb"},                         args = "[size] [transparency]", fn = function(v, v2)
		for _, c in ipairs(hitboxConns) do c:Disconnect() end
		hitboxConns = {}
		hitboxOrigSizes = {}
		if hitboxNccFolder then hitboxNccFolder:Destroy() end
		hitboxNccFolder = Instance.new("Folder", workspace)
		hitboxNccFolder.Name = "DeltaHitboxNCC"
		local size = tonumber(v) or 30
		local transp = tonumber(v2) or 0.9
		local ps = p.Parent
		local myHrp = p.Character and p.Character:FindFirstChild("HumanoidRootPart")
		local function applyNcc(hrp)
			if not myHrp then myHrp = p.Character and p.Character:FindFirstChild("HumanoidRootPart") end
			if myHrp then
				local ncc = Instance.new("NoCollisionConstraint", hitboxNccFolder)
				ncc.Part0 = myHrp
				ncc.Part1 = hrp
			end
		end
		local function monitor(hrp)
			local sz = Vector3.new(size, size, size)
			if not hitboxOrigSizes[hrp] then
				hitboxOrigSizes[hrp] = hrp.Size
			end
			local function apply()
				if hrp.Size.X ~= size then
					hrp.Size = sz
					hrp.Transparency = transp
				end
			end
			apply()
			applyNcc(hrp)
			hitboxConns[#hitboxConns+1] = hrp:GetPropertyChangedSignal("Size"):Connect(apply)
		end
		local function onChar(char)
			local hrp = char:WaitForChild("HumanoidRootPart", 5)
			if hrp then monitor(hrp) end
		end
		local function onPlayer(pl)
			if pl == p then return end
			if p.Team and pl.Team and pl.Team == p.Team then return end
			if pl.Character then onChar(pl.Character) end
			hitboxConns[#hitboxConns+1] = pl.CharacterAdded:Connect(onChar)
		end
		for _, pl in ipairs(ps:GetPlayers()) do onPlayer(pl) end
		hitboxConns[#hitboxConns+1] = ps.PlayerAdded:Connect(onPlayer)
		hitboxConns[#hitboxConns+1] = p.CharacterAdded:Connect(function(char)
			myHrp = char:WaitForChild("HumanoidRootPart", 5)
			if not myHrp then return end
			for _, pl in ipairs(ps:GetPlayers()) do
				if pl ~= p and pl.Character then
					local hrp = pl.Character:FindFirstChild("HumanoidRootPart")
					if hrp then applyNcc(hrp) end
				end
			end
		end)
	end },
	{ aliases = {"unhitbox", "unhb"},                     fn = function()

		hitboxConns = {}
		-- FIX: kembaliin ukuran asli semua HRP yang sudah diubah

				hrp.Size = origSize
				hrp.Transparency = 0
			end
		end
		hitboxOrigSizes = {}
		if hitboxNccFolder then hitboxNccFolder:Destroy(); hitboxNccFolder = nil end
	end },
